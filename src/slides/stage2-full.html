Use case: React's propTypes and defaultProps

class SimpleList extends React.Component {
  static propTypes = PropTypes.arrayOf(PropTypes.string)
  static defaultProps = { items: [] }
  render() {
    return <ul>
      { this.items.map(item => <li>{ item }</li>) }
    </ul>
  }
}
